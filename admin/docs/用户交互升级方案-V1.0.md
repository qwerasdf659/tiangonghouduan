export function animatedCounter(config = {}) {
  return {
    value: 0,
    targetValue: config.value || 0,
    suffix: config.suffix || '',
    prefix: config.prefix || '',
    decimals: config.decimals || 0,
    duration: config.duration || 1000,
    
    init() {
      // å»¶è¿Ÿå¯åŠ¨åŠ¨ç”»ï¼Œç­‰å¾…å…ƒç´ è¿›å…¥è§†å£
      this.$nextTick(() => {
        this.animateValue()
      })
    },
    
    animateValue() {
      const start = 0
      const end = this.targetValue
      const duration = this.duration
      const startTime = performance.now()
      
      const animate = (currentTime) => {
        const elapsed = currentTime - startTime
        const progress = Math.min(elapsed / duration, 1)
        
        // easeOutCubic ç¼“åŠ¨å‡½æ•° - å…ˆå¿«åæ…¢
        const easeProgress = 1 - Math.pow(1 - progress, 3)
        
        if (this.decimals > 0) {
          this.value = parseFloat((start + (end - start) * easeProgress).toFixed(this.decimals))
        } else {
          this.value = Math.floor(start + (end - start) * easeProgress)
        }
        
        if (progress < 1) {
          requestAnimationFrame(animate)
        } else {
          this.value = end
        }
      }
      
      requestAnimationFrame(animate)
    },
    
    // æ›´æ–°ç›®æ ‡å€¼ï¼ˆå“åº”å¼ï¼‰
    updateTarget(newValue) {
      this.targetValue = newValue
      this.animateValue()
    },
    
    get displayValue() {
      return this.prefix + this.formatNumber(this.value) + this.suffix
    },
    
    formatNumber(num) {
      if (this.decimals > 0) {
        return num.toLocaleString('zh-CN', { 
          minimumFractionDigits: this.decimals,
          maximumFractionDigits: this.decimals 
        })
      }
      return num.toLocaleString('zh-CN')
    }
  }
}
