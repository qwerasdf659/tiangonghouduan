/**
 * 🔄 积分管理API - 业务契约重定向层
 * 统一重定向到V4引擎的积分管理接口，保持API一致性和标准性
 *
 * 业务原则：
 * - API是业务契约，应该保持一致性和标准性
 * - 重定向到已验证的V4引擎实现，而非重复实现
 * - 基于真实数据验证的业务逻辑
 *
 * 创建时间：2025年09月18日 北京时间
 * 修复时间：2025年09月18日 北京时间 - 基于覆盖率分析修复

const express = require('express')
const router = express.Router()
const ApiResponse = require('../utils/ApiResponse')
 
/**
 * 🔄 用户积分增加 - 重定向到V4管理员接口
 * POST /api/points/add → POST /api/v4/unified-engine/admin/points/adjust
 *
 * 业务逻辑：使用V4引擎已验证的积分调整功能
 */
router.post('/add', async (req, res) => {
  try {
    const { userId, amount, reason } = req.body
 
    // 参数验证（与V4引擎保持一致）
    if (!userId || !amount || amount <= 0) {
      return res.status(400).json(
        ApiResponse.badRequest('参数错误：userId和amount必需，amount必须大于0')
      )
    }
 
    // 直接复制V4引擎的积分调整逻辑
    const models = require('../models')
 
    // 验证目标用户存在
    const targetUser = await models.User.findByPk(userId)
    if (!targetUser) {
      return res.status(400).json(
        ApiResponse.badRequest('目标用户不存在')
      )
    }
 
    // 获取用户积分账户
    let userPointsAccount = await models.UserPointsAccount.findOne({
      where: { user_id: userId }
    })
 
    if (!userPointsAccount) {
      // 创建新的积分账户
      userPointsAccount = await models.UserPointsAccount.create({
        user_id: userId,
        available_points: 0,
        total_earned: 0,
        total_consumed: 0,
        account_level: 1
      })
    }
 
    const oldPoints = parseFloat(userPointsAccount.available_points) || 0
    const newPoints = oldPoints + amount
 
    // 确保积分不为负数
    if (newPoints < 0) {
      return res.status(400).json(
        ApiResponse.badRequest('积分调整后不能为负数')
      )
    }
 
    // 更新积分账户
    await userPointsAccount.update({
      available_points: newPoints,
      updated_at: new Date()
    })
 
    // 创建积分变更记录
    await models.PointsTransaction.create({
      user_id: userId,
      account_id: userPointsAccount.account_id, // 🔧 修复：添加缺失的account_id
      transaction_type: 'earn',
      points_amount: amount, // 🔧 修复：使用正确字段名
      points_balance_before: oldPoints, // 🔧 修复：记录操作前余额
      points_balance_after: newPoints, // 🔧 修复：记录操作后余额
      business_type: 'admin_adjust', // 🔧 修复：设置业务类型
      source_type: 'api', // 🔧 添加：积分来源类型
      transaction_title: `API调整：${reason || '用户积分增加'}`, // 🔧 修复：使用正确字段名
      transaction_description: `管理员通过API为用户${userId}增加${amount}积分`, // 🔧 添加：详细描述
      operator_id: 1, // �� 修复：使用正确字段名（系统操作员）
      business_id: `api_add_${Date.now()}`, // 🔧 添加：业务关联ID
      trigger_event: 'api_points_add', // 🔧 添加：触发事件
      transaction_time: new Date(), // 🔧 添加：明确交易时间
      status: 'completed'
    })
 
    return res.status(200).json(
      ApiResponse.success({
        user_id: parseInt(userId), // 🔧 修复：确保返回数字类型
        old_points: oldPoints,
        adjustment: amount,
        new_points: newPoints,
        operation: 'add',
        reason: reason || '用户积分增加',
        adjusted_at: new Date().toISOString()
      }, '积分调整成功', 'POINTS_ADJUST_SUCCESS')
    )
  } catch (error) {
    console.error('积分增加失败:', error)
    return res.status(500).json(
      ApiResponse.serverError('积分增加失败', 'POINTS_ADD_ERROR', { error: error.message })
    )
  }
})

 *
 * 业务逻辑：使用V4引擎已验证的积分查询功能
 */
router.get('/balance/:userId', async (req, res) => {
  try {
    const { userId } = req.params
 
    if (!userId) {
      return res.status(400).json(
        ApiResponse.badRequest('缺少用户ID参数')
      )
    }
 
    // 直接调用V4引擎的积分查询功能
    const models = require('../models')
 
    const pointsAccount = await models.UserPointsAccount.findOne({
      where: { user_id: userId }
    })
 
    if (!pointsAccount) {
      return res.status(400).json(
        ApiResponse.badRequest('用户不存在或积分账户未创建')
      )
    }
 
    return res.status(200).json(
      ApiResponse.success({
        user_id: parseInt(userId), // 🔧 修复：确保返回数字类型
        available_points: pointsAccount.available_points,
        total_earned: pointsAccount.total_earned,
        total_spent: pointsAccount.total_spent,
        last_updated: pointsAccount.updated_at
      }, '获取用户积分成功', 'POINTS_QUERY_SUCCESS')
